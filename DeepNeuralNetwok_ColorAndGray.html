from tensorflow import keras
import tensorflow as tf
import numpy as np
import keras_preprocessing
from keras_preprocessing import image
from keras_preprocessing.image import ImageDataGenerator

train_data_path = '/content/drive/My Drive/DIP Project/Fruit-Images-Dataset-master/FinalColor/Training'
test_data_path = '/content/drive/My Drive/DIP Project/Fruit-Images-Dataset-master/FinalColor/Testing'

TrainDataGen = ImageDataGenerator(rescale = 1./255)
TestDataGen = ImageDataGenerator(rescale = 1./255)

TrainSet = TrainDataGen.flow_from_directory(train_data_path,target_size=(100,100),class_mode='categorical')
TestSet = TestDataGen.flow_from_directory(test_data_path,target_size=(100,100),class_mode='categorical')

dnn_model =  tf.keras.models.Sequential([
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(512, activation='relu'),
    tf.keras.layers.Dense(14, activation='softmax')
])

dnn_model.compile(loss = 'categorical_crossentropy', optimizer='rmsprop', metrics=['accuracy'])
history = dnn_model.fit_generator(TrainSet,epochs=25,validation_data = TestSet,verbose = 1)

import matplotlib.pyplot as plt
acc = history.history['acc']
val_acc = history.history['val_acc']
loss = history.history['loss']
val_loss = history.history['val_loss']
